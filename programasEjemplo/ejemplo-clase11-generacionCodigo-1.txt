dim global as real;

function doble(valor as integer) as integer
	return valor * 2;
end function;

proc main()
	dim local as integer;

	global = 1.0;
	local = 2;

	if global < 2.0 then
		print 1;	
	else
		print local;
	end if;

	print doble(2);
end proc;
